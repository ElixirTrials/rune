**Concurrent writes integration test:**
```python
def test_concurrent_writes_no_deadlock(tmp_path, make_adapter_record):
    import threading
    from sqlmodel import create_engine
    db_path = tmp_path / "concurrent.db"
    engine = create_engine(f"sqlite:///{db_path}")
    registry = AdapterRegistry(engine=engine)
    errors: list[Exception] = []

    def write(adapter_id: str) -> None:
        try:
            registry.store(make_adapter_record(id=adapter_id))
        except Exception as e:
            errors.append(e)

    threads = [threading.Thread(target=write, args=(f"t-{i}",)) for i in range(5)]
    for t in threads:
        t.start()
    for t in threads:
        t.join()

    assert errors == [], f"Concurrent writes produced errors: {errors}"
    assert len(registry.list_all()) == 5
```
